181 std::vector< std::size_t >
182 Carrier::queryBCs()
183 // *****************************************************************************
184 // Extract nodes IDs from side sets node lists and match to boundary conditions
185 //! \return List of owned node IDs on which a Dirichlet BC is set by the user
186 //! \details Boundary conditions, mathematically speaking, are applied on
187 //! finite surfaces. These finite surfaces are given by element sets. This
188 //! function takes the node lists mapped to side set IDs and extracts only
189 //! those nodes that we own that the user has specified boundary condition on
190 //! and returns the 'new' (as in producing contiguous-row-id linear system
191 //! contributions, see also Partitioner.h) node IDs.
192 //! \author J. Bakosi
193 // *****************************************************************************
194 {
195  // Access all side sets from LinSysMerger
196  auto& side = m_linsysmerger.ckLocalBranch()->side();
197 
198  // lambda to find out if we own the old (as in file) global node id. We
199  // perform a linear search on the map value so we do not have to invert the
200  // map and return the map key (the new global id, new as in producing
201  // contiguous-row-id linear system contributions, see also Partitioner.h).
202  auto own = [ this ]( std::size_t id ) -> std::pair< bool, std::size_t > {
203  for (const auto& i : this->m_cid)
204  if (i.second == id)
205  return { true, i.first };
206  return { false, 0 };
207  };
208 
209  // lambda to find out if the user has specified a Dirichlet BC on the given
210  // side set for any component of any of the PDEs integrated
211  auto userbc = []( int sideset ) -> bool {
212  for (const auto& eq : g_pdes) if (eq.anydirbc(sideset)) return true;
213  return false;
214  };
215 
216  // Collect list of owned node IDs on which a Dirichlet BC is given by the user
217  std::vector< std::size_t > bc;
218  for (const auto& s : side) {
219  auto ubc = userbc( s.first );
220  for (auto n : s.second) {
221  auto o = own(n);
222  if (o.first && ubc) bc.push_back( o.second );
223  }
224  }
225 
226  // Make BC node list unique. The node list extracted here can still contain
227  // repeating IDs, because the nodes are originally extracted by the Exodus
228  // reader by interrogating elements whose faces are adjacent to side sets,
229  // which can contain repeating node IDs for those nodes that elements share.
230  // See also the Exodus user manual.
231  tk::unique(bc);
232 
233  return bc;
234 }

289 void
290 Carrier::bcval( int frompe, const std::vector< std::size_t >& nodes )
291 // *****************************************************************************
292 // Look up boundary condition values at node IDs for all PDEs
293 //! \param[in] nodes Vector of node IDs at which to query BC values
294 //! \author J. Bakosi
295 // *****************************************************************************
296 {
297  // Access all side sets from LinSysMerger
298  auto& side = m_linsysmerger.ckLocalBranch()->side();
299 
300  // lambda to query the user-specified Dirichlet BCs on a given side set for
301  // all components of all the PDEs integrated
302  auto bc = []( int sideset ) -> std::vector< std::pair< bool, tk::real > > {
303  std::vector< std::pair< bool, tk::real > > b;
304  for (const auto& eq : g_pdes) {
305  auto e = eq.dirbc( sideset ); // query BC values for all components of eq
306  b.insert( end(b), begin(e), end(e) );
307  }
308  Assert( b.size() == g_inputdeck.get< tag::component >().nprop(),
309  "The total number of scalar components of all configured PDEs (" +
310  std::to_string( g_inputdeck.get< tag::component >().nprop() ) + ") "
311  "and the sum of the lengths of the Dirichlet BC vectors returned "
312  "from all configured PDE::dirbc() calls (" +
313  std::to_string( b.size() ) + ") does not match" );
314  return b;
315  };
316 
317  // lambda to find out whether 'new' node id is in the list of 'old' node ids
318  // given in s (which contains the old node ids of a side set). Here 'old'
319  // means as in file, while 'new' means as in producing contiguous-row-id
320  // linear system contributions. See also Partitioner.h.
321  auto inset = [ this ]( std::size_t id, const std::vector< std::size_t >& s )
322  -> bool {
323  for (auto n : s) if (tk::cref_find(this->m_cid,id) == n) return true;
324  return false;
325  };
326 
327  // Collect vector of pairs of bool and BC value, where the bool indicates
328  // whether the BC value is set at the given node by the user. The size of the
329  // vectors is the number of PDEs integrated times the number of scalar
330  // components in all PDEs. The vector is associated to global node IDs at
331  // which the boundary condition will be set. If a node gets boundary
332  // conditions from multiple side sets, true values (the fact that a BC is set)
333  // overwrite existing false ones, i.e., the union of the boundary conditions
334  // will be applied at the node.
335  std::unordered_map< std::size_t,
336  std::vector< std::pair< bool, tk::real > > > bcv;
337  for (const auto& s : side) {
338  auto b = bc( s.first ); // query BC values for all components of all PDEs
339  for (auto n : nodes)
340  if (inset( n, s.second )) {
341  auto& v = bcv[n];
342  v.resize( b.size() );
343  for (std::size_t i=0; i<b.size(); ++i) if (b[i].first) v[i] = b[i];
344  }
345  }
346 
347  m_linsysmerger[ frompe ].charebcval( bcv );
348 }

561 void
562 Carrier::genpar()
563 // *****************************************************************************
564 // Generate particles to each of our mesh cells
565 //! \author F.J. Gonzalez
566 // *****************************************************************************
567 {
568  auto rng = tk::RNGSSE< gm19_state, unsigned, gm19_generate_ >
569  ( static_cast<unsigned>(m_ncarr), gm19_init_sequence_ );
570 
571  // Create a reference of mesh point coordinates
572  const auto& x = m_coord[0];
573  const auto& y = m_coord[1];
574  const auto& z = m_coord[2];
575 
576  Assert( m_elp.size() >= m_particles.nunk(),
577  "Element-of-particle array not large enough" );
578 
579  // Generate npar number of particles into each mesh cell
580  auto npar = m_particles.nunk() / (m_inpoel.size()/4);
581  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
582  for (std::size_t p=0; p<npar; ++p) {
583  std::array< tk::real, 4 > N;
584  rng.uniform( thisIndex, 3, N.data() );
585  N[3] = 1.0 - N[0] - N[1] - N[2];
586  if ( std::min(N[0],1-N[0]) > 0 && std::min(N[1],1-N[1]) > 0 &&
587  std::min(N[2],1-N[2]) > 0 && std::min(N[3],1-N[3]) > 0 ) {
588  const auto A = m_inpoel[e*4+0];
589  const auto B = m_inpoel[e*4+1];
590  const auto C = m_inpoel[e*4+2];
591  const auto D = m_inpoel[e*4+3];
592  const auto i = e * npar + p;
593  m_particles(i,0,0) = x[A]*N[0] + x[B]*N[1] + x[C]*N[2] + x[D]*N[3];

603 Carrier::parinel( std::size_t p, std::size_t e, std::array< tk::real, 4 >& N )
604 // *****************************************************************************
605 // Search particle in a single mesh cell
606 //! \param[in] p Particle index
607 //! \param[in] e Mesh cell index
608 //! \param[inout] N Shapefunctions evaluated at the particle position
609 //! \return True if particle is in mesh cell
610 //! \author F.J. Gonzalez
611 // *****************************************************************************
612 {
613  // Tetrahedron node indices
614  const auto A = m_inpoel[e*4+0];
615  const auto B = m_inpoel[e*4+1];
616  const auto C = m_inpoel[e*4+2];
617  const auto D = m_inpoel[e*4+3];
618 
619  // Tetrahedron node coordinates
620  const auto& x = m_coord[0];
621  const auto& y = m_coord[1];
622  const auto& z = m_coord[2];
623 
624  // Particle coordinates
625  const auto& xp = m_particles( p, 0, 0 );
626  const auto& yp = m_particles( p, 1, 0 );
627  const auto& zp = m_particles( p, 2, 0 );
628 
629  // Evaluate linear shapefunctions at particle locations using Cramer's Rule
630  // | xp | | x1 x2 x3 x4 | | N1 |
631  // | yp | = | y1 y2 y3 y4 | • | N2 |
632  // | zp | | z1 z2 z3 z4 | | N3 |
633  // | 1 | | 1 1 1 1 | | N4 |
634 
635  tk::real DetX = (y[B]*z[C] - y[C]*z[B] - y[B]*z[D] + y[D]*z[B] +
636  y[C]*z[D] - y[D]*z[C])*x[A] + x[B]*y[C]*z[A] - x[B]*y[A]*z[C] +
637  x[C]*y[A]*z[B] - x[C]*y[B]*z[A] + x[B]*y[A]*z[D] - x[B]*y[D]*z[A] -
638  x[D]*y[A]*z[B] + x[D]*y[B]*z[A] - x[C]*y[A]*z[D] + x[C]*y[D]*z[A] +
639  x[D]*y[A]*z[C] - x[D]*y[C]*z[A] - x[B]*y[C]*z[D] + x[B]*y[D]*z[C] +
640  x[C]*y[B]*z[D] - x[C]*y[D]*z[B] - x[D]*y[B]*z[C] + x[D]*y[C]*z[B];
641 
642  tk::real DetX1 = (y[D]*z[C] - y[C]*z[D] + y[C]*zp - yp*z[C] -
643  y[D]*zp + yp*z[D])*x[B] + x[C]*y[B]*z[D] - x[C]*y[D]*z[B] -
644  x[D]*y[B]*z[C] + x[D]*y[C]*z[B] - x[C]*y[B]*zp + x[C]*yp*z[B] +
645  xp*y[B]*z[C] - xp*y[C]*z[B] + x[D]*y[B]*zp - x[D]*yp*z[B] -
646  xp*y[B]*z[D] + xp*y[D]*z[B] + x[C]*y[D]*zp - x[C]*yp*z[D] -
647  x[D]*y[C]*zp + x[D]*yp*z[C] + xp*y[C]*z[D] - xp*y[D]*z[C];
648 
649  tk::real DetX2 = (y[C]*z[D] - y[D]*z[C] - y[C]*zp + yp*z[C] +
650  y[D]*zp - yp*z[D])*x[A] + x[C]*y[D]*z[A] - x[C]*y[A]*z[D] +
651  x[D]*y[A]*z[C] - x[D]*y[C]*z[A] + x[C]*y[A]*zp - x[C]*yp*z[A] -
652  xp*y[A]*z[C] + xp*y[C]*z[A] - x[D]*y[A]*zp + x[D]*yp*z[A] +
653  xp*y[A]*z[D] - xp*y[D]*z[A] - x[C]*y[D]*zp + x[C]*yp*z[D] +
654  x[D]*y[C]*zp - x[D]*yp*z[C] - xp*y[C]*z[D] + xp*y[D]*z[C];
655 
656  tk::real DetX3 = (y[D]*z[B] - y[B]*z[D] + y[B]*zp - yp*z[B] -
657  y[D]*zp + yp*z[D])*x[A] + x[B]*y[A]*z[D] - x[B]*y[D]*z[A] -
658  x[D]*y[A]*z[B] + x[D]*y[B]*z[A] - x[B]*y[A]*zp + x[B]*yp*z[A] +
659  xp*y[A]*z[B] - xp*y[B]*z[A] + x[D]*y[A]*zp - x[D]*yp*z[A] -
660  xp*y[A]*z[D] + xp*y[D]*z[A] + x[B]*y[D]*zp - x[B]*yp*z[D] -
661  x[D]*y[B]*zp + x[D]*yp*z[B] + xp*y[B]*z[D] - xp*y[D]*z[B];
662 
663  tk::real DetX4 = (y[B]*z[C] - y[C]*z[B] - y[B]*zp + yp*z[B] +
664  y[C]*zp - yp*z[C])*x[A] + x[B]*y[C]*z[A] - x[B]*y[A]*z[C] +
665  x[C]*y[A]*z[B] - x[C]*y[B]*z[A] + x[B]*y[A]*zp - x[B]*yp*z[A] -
666  xp*y[A]*z[B] + xp*y[B]*z[A] - x[C]*y[A]*zp + x[C]*yp*z[A] +
667  xp*y[A]*z[C] - xp*y[C]*z[A] - x[B]*y[C]*zp + x[B]*yp*z[C] +
668  x[C]*y[B]*zp - x[C]*yp*z[B] - xp*y[B]*z[C] + xp*y[C]*z[B];
669 
670  // Shape functions evaluated at particle location
671  N[0] = DetX1/DetX;
672  N[1] = DetX2/DetX;
673  N[2] = DetX3/DetX;
674  N[3] = DetX4/DetX;
675 
676  // if min( N^i, 1-N^i ) > 0 for all i, particle is in cell
677  if ( std::min(N[0],1.0-N[0]) > 0 && std::min(N[1],1.0-N[1]) > 0 &&
678  std::min(N[2],1.0-N[2]) > 0 && std::min(N[3],1.0-N[3]) > 0 )
679  {
680  m_elp.resize( p+1 );
681  m_elp[ p ] = e;
682  return true;
683  } else {
684  return false;
685  }

719 void
720 Carrier::track()
721 // *****************************************************************************
722 // Advance our particles and search for their new mesh cells
723 //! \author F.J. Gonzalez
724 // *****************************************************************************
725 {
726  // Only advance particles in the finel time step stage
727  if (m_stage < 1) {
728  contribute( CkCallback( CkReductionTarget( Transporter, parcomcomplete ),
729  m_transporter));
730  return;
731  }
732 
733  // Lambda to attempt to find and advance particle i in element e. Returns true
734  // if the particle was found (and advanced), false if it was not found.
735  std::array< tk::real, 4 > N;
736  auto adv = [ this, &N ]( std::size_t i, std::size_t e ) -> bool {
737  if (this->parinel( i, e, N )) {
738  advanceParticle( i, e, N );
739  return true;
740  }
741  return false;
742  };
743 
744  // Search cells of our mesh chunk for all particles
745  for (std::size_t i=0; i<m_particles.nunk(); ++i) {
746  // First try to find particle where it has last been found
747  bool found = adv( i, m_elp[i] );
748  // Next search in the elements surroundings the points of the element where
749  // the particle has last been found
750  if (!found) {
751  auto last = m_esupel.second[m_elp[i]+1];
752  for (auto j=m_esupel.second[m_elp[i]]+1; j<=last; ++j) {
753  found = adv( i, m_esupel.first[j] );
754  if (found) j = last; // search for next particle
755  }
756  }
757  // Next search all cells in our chunk of the mesh
758  if (!found) {
759  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
760  found = adv( i, e );
761  if (found) e = m_inpoel.size()/4; // search for next particle
762  }
763  // If the particle still has not been found, it left our chunk of the
764  // mesh, mark as missing (will initiate communication to find it)
765  if (!found) m_parmiss.insert( i );
766  }
767  }
768 
769  // If we have no missing particles, we are done, if we do, send out requests
770  // to find them to those Carrier chares with which we neighbor mesh cells
771  if (m_parmiss.empty()) {
772  m_nchpar = 0;
773  m_parmiss.clear();
774  m_parelse.clear();
775  contribute( CkCallback( CkReductionTarget( Transporter, parcomcomplete ),
776  m_transporter));
777  } else {
778  decltype(m_particles) pexp( m_parmiss.size(), 3 );
779  std::size_t p = 0;
780  for (auto i : m_parmiss) {
781  pexp( p, 0, 0 ) = m_particles( i, 0, 0 );
782  pexp( p, 1, 0 ) = m_particles( i, 1, 0 );
783  pexp( p, 2, 0 ) = m_particles( i, 2, 0 );
784  ++p;
785  }
786  std::vector< std::size_t > miss( begin(m_parmiss), end(m_parmiss) );
787  for (auto n : m_msum) thisProxy[ n ].findpar( thisIndex, miss, pexp );
788  }
789 }

898 void
899 Carrier::advanceParticle( std::size_t i,
900  std::size_t e,
901  const std::array< tk::real, 4>& N )
902 // *****************************************************************************
903 // Advance particle based on velocity from mesh cell
904 //! \author F.J. Gonzalez
905 // *****************************************************************************
906 {
907  auto dt = g_inputdeck.get< tag::discr, tag::dt >();
908  const auto A = m_inpoel[e*4+0];
909  const auto B = m_inpoel[e*4+1];
910  const auto C = m_inpoel[e*4+2];
911  const auto D = m_inpoel[e*4+3];
912 
913  // Extract velocity at the four cell nodes at current and previous time step.
914  // To keep the code below as general as possible, we interrogate all PDEs
915  // configured and use the last nonzero velocity vector.
916  std::vector< std::array< tk::real, 4 > > c, p;
917  for (const auto& eq : g_pdes) {
918  auto v = eq.velocity( m_u, A, B, C, D );
919  if (!v.empty()) c = std::move(v);
920  auto w = eq.velocity( m_up, A, B, C, D );
921  if (!w.empty()) p = std::move(w);
922  }
923 
924  Assert(c.size() == 3 && p.size() == 3, "PDE velocity must have 3 components");
925 
926  // Compute velocity differnce between current and previous time steps
927  const auto& cx = c[0];
928  const auto& px = p[0];
929  const auto& cy = c[1];
930  const auto& py = p[1];
931  const auto& cz = c[2];
932  const auto& pz = p[2];
933  tk::real dvx[4] = { cx[0]-px[0], cx[1]-px[1], cx[2]-px[2], cx[3]-px[3] };
934  tk::real dvy[4] = { cy[0]-py[0], cy[1]-py[1], cy[2]-py[2], cy[3]-py[3] };
935  tk::real dvz[4] = { cz[0]-pz[0], cz[1]-pz[1], cz[2]-pz[2], cz[3]-pz[3] };
936 
937  // Advance particle coordinates using the interpolated velocity
938  m_particles( i, 0, 0 ) +=
939  dt*(N[0]*dvx[0] + N[1]*dvx[1] + N[2]*dvx[2] + N[3]*dvx[3]);
940  m_particles( i, 1, 0 ) +=
941  dt*(N[0]*dvy[0] + N[1]*dvy[1] + N[2]*dvy[2] + N[3]*dvy[3]);
942  m_particles( i, 2, 0 ) +=
943  dt*(N[0]*dvz[0] + N[1]*dvz[1] + N[2]*dvz[2] + N[3]*dvz[3]);
944 
945  // Apply boundary conditions to particle
946  applyParBC( i );
947 }
