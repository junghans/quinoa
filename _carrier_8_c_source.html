133 Carrier::comm()
134 // *****************************************************************************
135 // Starts collecting global mesh node IDs bordering the mesh chunk held by
136 // fellow Carrier chares associated to their chare IDs
137 //! \author J. Bakosi
138 // *****************************************************************************
139 {
140  // Read coordinates of nodes of the mesh chunk we operate on
141  readCoords();
142 
143  const auto& x = m_coord[0];
144  const auto& y = m_coord[1];
145  const auto& z = m_coord[2];
146 
147  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
148  const std::array< std::size_t, 4 > N{{ m_inpoel[e*4+0], m_inpoel[e*4+1],
149  m_inpoel[e*4+2], m_inpoel[e*4+3] }};
150  // compute element Jacobi determinant * 5/120 = element volume / 4
151  const std::array< tk::real, 3 >
152  ba{{ x[N[1]]-x[N[0]], y[N[1]]-y[N[0]], z[N[1]]-z[N[0]] }},
153  ca{{ x[N[2]]-x[N[0]], y[N[2]]-y[N[0]], z[N[2]]-z[N[0]] }},
154  da{{ x[N[3]]-x[N[0]], y[N[3]]-y[N[0]], z[N[3]]-z[N[0]] }};
155  const auto J = tk::triple( ba, ca, da ) * 5.0 / 120.0;
156  // scatter add V/4 to nodes
157  for (std::size_t j=0; j<4; ++j) m_vol[N[j]] += J;
158  }
159 
160  // Collect fellow chare IDs (as well as the global mesh node IDs and the
161  // volume associated to them) to/from our mesh neighbors. The goal here is to
162  // end up with a list of chare IDs (as the keys in m_msum) on each chare that
163  // the chare shares at least a single mesh node ID with, i.e., its mesh chunk
164  // borders another chare's mesh chunk. Note that this is not necessarily the
165  // most efficient way of doing this, since we do an all-to-all reduction from
166  // and to all Carrier chares, sending all the global node indices of the mesh
167  // chunk each Carrier chare holds. This contribute() is the start of this, and
168  // msum() is the final step. A better way of doing this is to only send the
169  // chare-boundary node indices. That information might already be available in
170  // Partitioner (the host chare group we are created by), or if not directly,
171  // it might be possible to be computed from the output of the partitioner.
172  // This data structure (m_msum) is used in various ways: (1) the list of
173  // neighboring chare IDs are used for searching for particles traveling across
174  // the mesh that left our mesh chunk, (2) the list of neighboring mesh node
175  // IDs (that lie on the chare-boundary) and their associated volumes are used
176  // by the FCT algorithm.
177  std::vector< std::pair< int, comm_t > >
178  meshnodes{ { thisIndex, std::make_tuple( m_gid, m_vol ) } };
179  auto stream = serialize( meshnodes );
180  CkCallback cb( CkIndex_Carrier::msum(nullptr), thisProxy );
181  contribute( stream.first, stream.second.get(), FieldsMerger, cb );
182 }

270 std::vector< std::size_t >
271 Carrier::queryBCs()
272 // *****************************************************************************
273 // Extract nodes IDs from side sets node lists and match to boundary conditions
274 //! \return List of owned node IDs on which a Dirichlet BC is set by the user
275 //! \details Boundary conditions, mathematically speaking, are applied on
276 //! finite surfaces. These finite surfaces are given by element sets. This
277 //! function takes the node lists mapped to side set IDs and extracts only
278 //! those nodes that we own that the user has specified boundary condition on
279 //! and returns the 'new' (as in producing contiguous-row-id linear system
280 //! contributions, see also Partitioner.h) node IDs.
281 //! \author J. Bakosi
282 // *****************************************************************************
283 {
284  // Access all side sets from LinSysMerger
285  auto& side = m_linsysmerger.ckLocalBranch()->side();
286 
287  // lambda to find out if we own the old (as in file) global node id. We
288  // perform a linear search on the map value so we do not have to invert the
289  // map and return the map key (the new global id, new as in producing
290  // contiguous-row-id linear system contributions, see also Partitioner.h).
291  auto own = [ this ]( std::size_t id ) -> std::pair< bool, std::size_t > {
292  for (const auto& i : this->m_cid)
293  if (i.second == id)
294  return { true, i.first };
295  return { false, 0 };
296  };
297 
298  // lambda to find out if the user has specified a Dirichlet BC on the given
299  // side set for any component of any of the PDEs integrated
300  auto userbc = []( int sideset ) -> bool {
301  for (const auto& eq : g_pdes) if (eq.anydirbc(sideset)) return true;
302  return false;
303  };
304 
305  // Collect list of owned node IDs on which a Dirichlet BC is given by the user
306  std::vector< std::size_t > bc;
307  for (const auto& s : side) {
308  auto ubc = userbc( s.first );
309  for (auto n : s.second) {
310  auto o = own(n);
311  if (o.first && ubc) bc.push_back( o.second );
312  }
313  }
314 
315  // Make BC node list unique. The node list extracted here can still contain
316  // repeating IDs, because the nodes are originally extracted by the Exodus
317  // reader by interrogating elements whose faces are adjacent to side sets,
318  // which can contain repeating node IDs for those nodes that elements share.
319  // See also the Exodus user manual.
320  tk::unique(bc);
321 
322  return bc;
323 }

378 void
379 Carrier::bcval( int frompe, const std::vector< std::size_t >& nodes )
380 // *****************************************************************************
381 // Look up boundary condition values at node IDs for all PDEs
382 //! \param[in] nodes Vector of node IDs at which to query BC values
383 //! \author J. Bakosi
384 // *****************************************************************************
385 {
386  // Access all side sets from LinSysMerger
387  auto& side = m_linsysmerger.ckLocalBranch()->side();
388 
389  // lambda to query the user-specified Dirichlet BCs on a given side set for
390  // all components of all the PDEs integrated
391  auto bc = []( int sideset ) -> std::vector< std::pair< bool, tk::real > > {
392  std::vector< std::pair< bool, tk::real > > b;
393  for (const auto& eq : g_pdes) {
394  auto e = eq.dirbc( sideset ); // query BC values for all components of eq
395  b.insert( end(b), begin(e), end(e) );
396  }
397  Assert( b.size() == g_inputdeck.get< tag::component >().nprop(),
398  "The total number of scalar components of all configured PDEs (" +
399  std::to_string( g_inputdeck.get< tag::component >().nprop() ) + ") "
400  "and the sum of the lengths of the Dirichlet BC vectors returned "
401  "from all configured PDE::dirbc() calls (" +
402  std::to_string( b.size() ) + ") does not match" );
403  return b;
404  };
405 
406  // lambda to find out whether 'new' node id is in the list of 'old' node ids
407  // given in s (which contains the old node ids of a side set). Here 'old'
408  // means as in file, while 'new' means as in producing contiguous-row-id
409  // linear system contributions. See also Partitioner.h.
410  auto inset = [ this ]( std::size_t id, const std::vector< std::size_t >& s )
411  -> bool {
412  for (auto n : s) if (tk::cref_find(this->m_cid,id) == n) return true;
413  return false;
414  };
415 
416  // Collect vector of pairs of bool and BC value, where the bool indicates
417  // whether the BC value is set at the given node by the user. The size of the
418  // vectors is the number of PDEs integrated times the number of scalar
419  // components in all PDEs. The vector is associated to global node IDs at
420  // which the boundary condition will be set. If a node gets boundary
421  // conditions from multiple side sets, true values (the fact that a BC is set)
422  // overwrite existing false ones, i.e., the union of the boundary conditions
423  // will be applied at the node.
424  std::unordered_map< std::size_t,
425  std::vector< std::pair< bool, tk::real > > > bcv;
426  for (const auto& s : side) {
427  auto b = bc( s.first ); // query BC values for all components of all PDEs
428  for (auto n : nodes)
429  if (inset( n, s.second )) {
430  auto& v = bcv[n];
431  v.resize( b.size() );
432  for (std::size_t i=0; i<b.size(); ++i) if (b[i].first) v[i] = b[i];
433  }
434  }
435 
436  m_linsysmerger[ frompe ].charebcval( bcv );
437 }

489 void
490 Carrier::rhs( tk::real mult, tk::real dt, const tk::Fields& sol )
491 // *****************************************************************************
492 // Compute right-hand side of transport equations
493 //! \param[in] mult Multiplier differentiating the different stages in
494 //! multi-stage time stepping
495 //! \param[in] dt Size of time step
496 //! \param[in] sol Solution vector at current stage
497 //! \author J. Bakosi
498 // *****************************************************************************
499 {
500  // Initialize FCT data structures for new time step stage
501  m_p.fill( 0.0 );
502  m_a.fill( 0.0 );
503  for (std::size_t p=0; p<m_u.nunk(); ++p)
504  for (ncomp_t c=0; c<g_inputdeck.get<tag::component>().nprop(); ++c) {
505  m_q(p,c*2+0,0) = -std::numeric_limits< tk::real >::max();
506  m_q(p,c*2+1,0) = std::numeric_limits< tk::real >::max();
507  }
508 
509  for (auto& b : m_pc) std::fill( begin(b), end(b), 0.0 );
510  for (auto& b : m_ac) std::fill( begin(b), end(b), 0.0 );
511  for (auto& b : m_qc)
512  for (ncomp_t c=0; c<m_u.nprop(); ++c) {
513  b[c*2+0] = -std::numeric_limits< tk::real >::max();
514  b[c*2+1] = std::numeric_limits< tk::real >::max();
515  }
516 
517  // Compute right-hand side vector for all equations
518  tk::Fields r( m_gid.size(), g_inputdeck.get< tag::component >().nprop() );
519  for (const auto& eq : g_pdes) eq.rhs( mult, dt, m_coord, m_inpoel, sol, r );
520  // Send off right-hand sides for assembly
521  m_linsysmerger.ckLocalBranch()->charerhs( thisIndex, m_gid, r );
522 
523  // Compute mass diffusion rhs contribution required for the low order solution
524  auto diff = m_fluxcorrector.diff( m_coord, m_inpoel, sol );
525  // Send off mass diffusion rhs contribution for assembly
526  m_linsysmerger.ckLocalBranch()->charediff( thisIndex, m_gid, diff );
527 }

787 //! the limited antidiffusive element contributions assembled to nodes
788 //! (Lohner: AEC^c), see also FluxCorrector::limit().

882 void
883 Carrier::genpar()
884 // *****************************************************************************
885 // Generate particles to each of our mesh cells
886 //! \author F.J. Gonzalez
887 // *****************************************************************************
888 {
889  auto rng = tk::RNGSSE< gm19_state, unsigned, gm19_generate_ >
890  ( static_cast<unsigned>(m_ncarr), gm19_init_sequence_ );
891 
892  // Create a reference of mesh point coordinates
893  const auto& x = m_coord[0];
894  const auto& y = m_coord[1];
895  const auto& z = m_coord[2];
896 
897  Assert( m_elp.size() >= m_particles.nunk(),
898  "Element-of-particle array not large enough" );
899 
900  // Generate npar number of particles into each mesh cell
901  auto npar = m_particles.nunk() / (m_inpoel.size()/4);
902  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
903  for (std::size_t p=0; p<npar; ++p) {
904  std::array< tk::real, 4 > N;
905  rng.uniform( thisIndex, 3, N.data() );
906  N[3] = 1.0 - N[0] - N[1] - N[2];
907  if ( std::min(N[0],1-N[0]) > 0 && std::min(N[1],1-N[1]) > 0 &&
908  std::min(N[2],1-N[2]) > 0 && std::min(N[3],1-N[3]) > 0 ) {
909  const auto A = m_inpoel[e*4+0];
910  const auto B = m_inpoel[e*4+1];
911  const auto C = m_inpoel[e*4+2];
912  const auto D = m_inpoel[e*4+3];
913  const auto i = e * npar + p;
914  m_particles(i,0,0) = x[A]*N[0] + x[B]*N[1] + x[C]*N[2] + x[D]*N[3];
915  m_particles(i,1,0) = y[A]*N[0] + y[B]*N[1] + y[C]*N[2] + y[D]*N[3];
916  m_particles(i,2,0) = z[A]*N[0] + z[B]*N[1] + z[C]*N[2] + z[D]*N[3];
917  m_elp[i] = e;
918  } else --p; // retry if particle was not generated into cell
919  }
920  }

923 bool
924 Carrier::parinel( std::size_t p, std::size_t e, std::array< tk::real, 4 >& N )
925 // *****************************************************************************
926 // Search particle in a single mesh cell
927 //! \param[in] p Particle index
928 //! \param[in] e Mesh cell index
929 //! \param[inout] N Shapefunctions evaluated at the particle position
930 //! \return True if particle is in mesh cell
931 //! \author F.J. Gonzalez
932 // *****************************************************************************
933 {
934  // Tetrahedron node indices
935  const auto A = m_inpoel[e*4+0];
936  const auto B = m_inpoel[e*4+1];
937  const auto C = m_inpoel[e*4+2];
938  const auto D = m_inpoel[e*4+3];
939 
940  // Tetrahedron node coordinates
941  const auto& x = m_coord[0];
942  const auto& y = m_coord[1];
943  const auto& z = m_coord[2];
944 
945  // Particle coordinates
946  const auto& xp = m_particles( p, 0, 0 );
947  const auto& yp = m_particles( p, 1, 0 );
948  const auto& zp = m_particles( p, 2, 0 );
949 
950  // Evaluate linear shapefunctions at particle locations using Cramer's Rule
951  // | xp | | x1 x2 x3 x4 | | N1 |
952  // | yp | = | y1 y2 y3 y4 | • | N2 |
953  // | zp | | z1 z2 z3 z4 | | N3 |
954  // | 1 | | 1 1 1 1 | | N4 |
955 
956  tk::real DetX = (y[B]*z[C] - y[C]*z[B] - y[B]*z[D] + y[D]*z[B] +
957  y[C]*z[D] - y[D]*z[C])*x[A] + x[B]*y[C]*z[A] - x[B]*y[A]*z[C] +
958  x[C]*y[A]*z[B] - x[C]*y[B]*z[A] + x[B]*y[A]*z[D] - x[B]*y[D]*z[A] -
959  x[D]*y[A]*z[B] + x[D]*y[B]*z[A] - x[C]*y[A]*z[D] + x[C]*y[D]*z[A] +
960  x[D]*y[A]*z[C] - x[D]*y[C]*z[A] - x[B]*y[C]*z[D] + x[B]*y[D]*z[C] +
961  x[C]*y[B]*z[D] - x[C]*y[D]*z[B] - x[D]*y[B]*z[C] + x[D]*y[C]*z[B];
962 
963  tk::real DetX1 = (y[D]*z[C] - y[C]*z[D] + y[C]*zp - yp*z[C] -
964  y[D]*zp + yp*z[D])*x[B] + x[C]*y[B]*z[D] - x[C]*y[D]*z[B] -
965  x[D]*y[B]*z[C] + x[D]*y[C]*z[B] - x[C]*y[B]*zp + x[C]*yp*z[B] +
966  xp*y[B]*z[C] - xp*y[C]*z[B] + x[D]*y[B]*zp - x[D]*yp*z[B] -
967  xp*y[B]*z[D] + xp*y[D]*z[B] + x[C]*y[D]*zp - x[C]*yp*z[D] -
968  x[D]*y[C]*zp + x[D]*yp*z[C] + xp*y[C]*z[D] - xp*y[D]*z[C];
969 
970  tk::real DetX2 = (y[C]*z[D] - y[D]*z[C] - y[C]*zp + yp*z[C] +
971  y[D]*zp - yp*z[D])*x[A] + x[C]*y[D]*z[A] - x[C]*y[A]*z[D] +
972  x[D]*y[A]*z[C] - x[D]*y[C]*z[A] + x[C]*y[A]*zp - x[C]*yp*z[A] -
973  xp*y[A]*z[C] + xp*y[C]*z[A] - x[D]*y[A]*zp + x[D]*yp*z[A] +
974  xp*y[A]*z[D] - xp*y[D]*z[A] - x[C]*y[D]*zp + x[C]*yp*z[D] +
975  x[D]*y[C]*zp - x[D]*yp*z[C] - xp*y[C]*z[D] + xp*y[D]*z[C];
976 
977  tk::real DetX3 = (y[D]*z[B] - y[B]*z[D] + y[B]*zp - yp*z[B] -
978  y[D]*zp + yp*z[D])*x[A] + x[B]*y[A]*z[D] - x[B]*y[D]*z[A] -
979  x[D]*y[A]*z[B] + x[D]*y[B]*z[A] - x[B]*y[A]*zp + x[B]*yp*z[A] +
980  xp*y[A]*z[B] - xp*y[B]*z[A] + x[D]*y[A]*zp - x[D]*yp*z[A] -
981  xp*y[A]*z[D] + xp*y[D]*z[A] + x[B]*y[D]*zp - x[B]*yp*z[D] -
982  x[D]*y[B]*zp + x[D]*yp*z[B] + xp*y[B]*z[D] - xp*y[D]*z[B];
983 
984  tk::real DetX4 = (y[B]*z[C] - y[C]*z[B] - y[B]*zp + yp*z[B] +
985  y[C]*zp - yp*z[C])*x[A] + x[B]*y[C]*z[A] - x[B]*y[A]*z[C] +
986  x[C]*y[A]*z[B] - x[C]*y[B]*z[A] + x[B]*y[A]*zp - x[B]*yp*z[A] -
987  xp*y[A]*z[B] + xp*y[B]*z[A] - x[C]*y[A]*zp + x[C]*yp*z[A] +
988  xp*y[A]*z[C] - xp*y[C]*z[A] - x[B]*y[C]*zp + x[B]*yp*z[C] +
989  x[C]*y[B]*zp - x[C]*yp*z[B] - xp*y[B]*z[C] + xp*y[C]*z[B];
990 
991  // Shape functions evaluated at particle location
992  N[0] = DetX1/DetX;
993  N[1] = DetX2/DetX;
994  N[2] = DetX3/DetX;
995  N[3] = DetX4/DetX;
996 
997  // if min( N^i, 1-N^i ) > 0 for all i, particle is in cell
998  if ( std::min(N[0],1.0-N[0]) > 0 && std::min(N[1],1.0-N[1]) > 0 &&
999  std::min(N[2],1.0-N[2]) > 0 && std::min(N[3],1.0-N[3]) > 0 )
1000  {
1001  m_elp.resize( p+1 );
1002  m_elp[ p ] = e;
1003  return true;
1004  } else {
1005  return false;
1006  }
1007 }

1041 Carrier::track()

1043 // Advance our particles and search for their new mesh cells
1044 //! \author F.J. Gonzalez
1045 // *****************************************************************************
1046 {
1047  // Only advance particles in the finel time step stage
1048  if (m_stage < 1) {
1049  contribute( CkCallback( CkReductionTarget( Transporter, parcomcomplete ),
1050  m_transporter));
1051  return;
1052  }
1053 
1054  // Lambda to attempt to find and advance particle i in element e. Returns true
1055  // if the particle was found (and advanced), false if it was not found.
1056  std::array< tk::real, 4 > N;
1057  auto adv = [ this, &N ]( std::size_t i, std::size_t e ) -> bool {
1058  if (this->parinel( i, e, N )) {
1059  advanceParticle( i, e, N );
1060  return true;
1061  }
1062  return false;
1063  };
1064 
1065  // Search cells of our mesh chunk for all particles
1066  for (std::size_t i=0; i<m_particles.nunk(); ++i) {
1067  // First try to find particle where it has last been found
1068  bool found = adv( i, m_elp[i] );
1069  // Next search in the elements surroundings the points of the element where
1070  // the particle has last been found
1071  if (!found) {
1072  auto last = m_esupel.second[m_elp[i]+1];
1073  for (auto j=m_esupel.second[m_elp[i]]+1; j<=last; ++j) {
1074  found = adv( i, m_esupel.first[j] );
1075  if (found) j = last; // search for next particle
1076  }
1077  }
1078  // Next search all cells in our chunk of the mesh
1079  if (!found) {
1080  for (std::size_t e=0; e<m_inpoel.size()/4; ++e) {
1081  found = adv( i, e );
1082  if (found) e = m_inpoel.size()/4; // search for next particle
1083  }
1084  // If the particle still has not been found, it left our chunk of the
1085  // mesh, mark as missing (will initiate communication to find it)
1086  if (!found) m_parmiss.insert( i );
1087  }
1088  }
1089 
1090  // If we have no missing particles, we are done, if we do, send out requests
1091  // to find them to those Carrier chares with which we neighbor mesh cells
1092  if (m_parmiss.empty()) {
1093  m_nchpar = 0;
1094  m_parmiss.clear();
1095  m_parelse.clear();
1096  contribute( CkCallback( CkReductionTarget( Transporter, parcomcomplete ),
1097  m_transporter));
1098  } else {
1099  decltype(m_particles) pexp( m_parmiss.size(), 3 );
1100  std::size_t p = 0;
1101  for (auto i : m_parmiss) {
1102  pexp( p, 0, 0 ) = m_particles( i, 0, 0 );
1103  pexp( p, 1, 0 ) = m_particles( i, 1, 0 );
1104  pexp( p, 2, 0 ) = m_particles( i, 2, 0 );
1105  ++p;
1106  }
1107  std::vector< std::size_t > miss( begin(m_parmiss), end(m_parmiss) );
1108  for (const auto& n : m_msum)
1109  thisProxy[ n.first ].findpar( thisIndex, miss, pexp );
1110  }

1220 void
1221 Carrier::advanceParticle( std::size_t i,
1222  std::size_t e,
1223  const std::array< tk::real, 4 >& Np )
1224 // *****************************************************************************
1225 // Advance particle based on velocity from mesh cell
1226 //! \author F.J. Gonzalez
1227 // *****************************************************************************
1228 {
1229  auto dt = g_inputdeck.get< tag::discr, tag::dt >();
1230 
1231  // Extract velocity at the four cell nodes at current and previous time step.
1232  // To keep the code below as general as possible, we interrogate all PDEs
1233  // configured and use the last nonzero velocity vector.
1234  std::vector< std::array< tk::real, 4 > > c, p;
1235  for (const auto& eq : g_pdes) {
1236  const std::array< std::size_t, 4 > N{{ m_inpoel[e*4+0], m_inpoel[e*4+1],
1237  m_inpoel[e*4+2], m_inpoel[e*4+3] }};
1238  auto v = eq.velocity( m_u, m_coord, N );
1239  if (!v.empty()) c = std::move(v);
1240  auto w = eq.velocity( m_up, m_coord, N );
1241  if (!w.empty()) p = std::move(w);
1242  }
1243 
1244  Assert(c.size() == 3 && p.size() == 3, "PDE velocity must have 3 components");
1245 
1246  // Compute velocity differnce between current and previous time steps
1247  const auto& cx = c[0];
1248  const auto& px = p[0];
1249  const auto& cy = c[1];
1250  const auto& py = p[1];
1251  const auto& cz = c[2];
1252  const auto& pz = p[2];
1253  tk::real dvx[4] = { cx[0]-px[0], cx[1]-px[1], cx[2]-px[2], cx[3]-px[3] };
1254  tk::real dvy[4] = { cy[0]-py[0], cy[1]-py[1], cy[2]-py[2], cy[3]-py[3] };
1255  tk::real dvz[4] = { cz[0]-pz[0], cz[1]-pz[1], cz[2]-pz[2], cz[3]-pz[3] };
1256 
1257  // Advance particle coordinates using the interpolated velocity
1258  m_particles( i, 0, 0 ) +=
1259  dt*(Np[0]*dvx[0] + Np[1]*dvx[1] + Np[2]*dvx[2] + Np[3]*dvx[3]);
1260  m_particles( i, 1, 0 ) +=
1261  dt*(Np[0]*dvy[0] + Np[1]*dvy[1] + Np[2]*dvy[2] + Np[3]*dvy[3]);
1262  m_particles( i, 2, 0 ) +=
1263  dt*(Np[0]*dvz[0] + Np[1]*dvz[1] + Np[2]*dvz[2] + Np[3]*dvz[3]);
1264 
1265  // Apply boundary conditions to particle
1266  applyParBC( i );
1267 }
