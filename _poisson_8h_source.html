63  //! Compute the left hand side sparse matrix
64  //! \param[in] coord Mesh node coordinates
65  //! \param[in] inpoel Mesh element connectivity
66  //! \param[in] psup Linked lists storing IDs of points surrounding points
67  //! \param[in,out] lhsd Diagonal of the sparse matrix storing nonzeros
68  //! \param[in,out] lhso Off-diagonal of the sparse matrix storing nonzeros
69  //! \details Sparse matrix storing the nonzero matrix values at rows and
70  //! columns given by psup. The format is similar to compressed row
71  //! storage, but the diagonal and off-diagonal data are stored in separate
72  //! vectors. For the off-diagonal data the local row and column indices,
73  //! at which values are nonzero, are stored by psup (psup1 and psup2,
74  //! where psup2 holds the indices at which psup1 holds the point ids
75  //! surrounding points, see also tk::genPsup()). Note that the number of
76  //! mesh points (our chunk) npoin = psup.second.size()-1.
77  //! \author J. Bakosi
78  void lhs( const std::array< std::vector< tk::real >, 3 >& coord,
79  const std::vector< std::size_t >& inpoel,
80  const std::pair< std::vector< std::size_t >,
81  std::vector< std::size_t > >& psup,
82  tk::MeshNodes& lhsd,
83  tk::MeshNodes& lhso ) const
84  {
85  Assert( psup.second.size()-1 == coord[0].size(),
86  "Number of mesh points and number of global IDs unequal" );
87  Assert( lhsd.nunk() == psup.second.size()-1, "Number of unknowns in "
88  "diagonal sparse matrix storage incorrect" );
89  Assert( lhso.nunk() == psup.first.size(), "Number of unknowns in "
90  "off-diagonal sparse matrix storage incorrect" );
91 
92  // Lambda to compute the sparse matrix vector index for row and column
93  // indices. Used only for off-diagonal entries.
94  auto spidx = [ &psup ]( std::size_t r, std::size_t c ) -> std::size_t {
95  Assert( r != c, "Only for computing the off-diagonal indices" );
96  for (auto i=psup.second[r]+1; i<=psup.second[r+1]; ++i)
97  if (c == psup.first[i]) return i;
98  Throw( "Cannot find row, column: " + std::to_string(r) + ',' +
99  std::to_string(c) + " in sparse matrix" );
100  };
101 
102  const auto& x = coord[0];
103  const auto& y = coord[1];
104  const auto& z = coord[2];
105 
106  // Zero matrix for all components
107  for (ncomp_t c=0; c<m_ncomp; ++c) {
108  lhsd.fill( c, m_offset, 0.0 );
109  lhso.fill( c, m_offset, 0.0 );
110  }
111 
112  for (std::size_t e=0; e<inpoel.size()/4; ++e) {
113  const auto A = inpoel[e*4+0];
114  const auto B = inpoel[e*4+1];
115  const auto C = inpoel[e*4+2];
116  const auto D = inpoel[e*4+3];
117  std::array< tk::real, 3 > ba{{ x[B]-x[A], y[B]-y[A], z[B]-z[A] }},
118  ca{{ x[C]-x[A], y[C]-y[A], z[C]-z[A] }},
119  da{{ x[D]-x[A], y[D]-y[A], z[D]-z[A] }};
120  const auto J = tk::triple( ba, ca, da );
121 
122  // shape function derivatives
123  std::array< std::array< tk::real, 3 >, 4 > grad; // nnode*ndim [4][3]
124  grad[1] = tk::crossdiv( ca, da, J );
125  grad[2] = tk::crossdiv( da, ba, J );
126  grad[3] = tk::crossdiv( ba, ca, J );
127  for (std::size_t i=0; i<3; ++i)
128  grad[0][i] = -grad[1][i]-grad[2][i]-grad[3][i];
129 
130  const auto iab = spidx(A,B);
131  const auto iac = spidx(A,C);
132  const auto iad = spidx(A,D);
133  const auto iba = spidx(B,A);
134  const auto ibc = spidx(B,C);
135  const auto ibd = spidx(B,D);
136  const auto ica = spidx(C,A);
137  const auto icb = spidx(C,B);
138  const auto icd = spidx(C,D);
139  const auto ida = spidx(D,A);
140  const auto idb = spidx(D,B);
141  const auto idc = spidx(D,C);
142 
143  for (ncomp_t c=0; c<m_ncomp; ++c) {
144  const auto r = lhsd.cptr( c, m_offset );
145  const auto s = lhso.cptr( c, m_offset );
146 
147  for (std::size_t i=0; i<4; ++i)
148  for (std::size_t j=0; j<4; ++j)
149  for (std::size_t k=0; k<3; ++k) {
150  const auto val = J/6.0 * grad[i][k] * grad[j][k];
151  if (i==j) {
152  lhsd.var( r, A ) -= val;
153  lhsd.var( r, B ) -= val;
154  lhsd.var( r, C ) -= val;
155  lhsd.var( r, D ) -= val;
156  } else {
157  lhso.var( s, iab ) -= val;
158  lhso.var( s, iac ) -= val;
159  lhso.var( s, iad ) -= val;
160  lhso.var( s, iba ) -= val;
161  lhso.var( s, ibc ) -= val;
162  lhso.var( s, ibd ) -= val;
163  lhso.var( s, ica ) -= val;
164  lhso.var( s, icb ) -= val;
165  lhso.var( s, icd ) -= val;
166  lhso.var( s, ida ) -= val;
167  lhso.var( s, idb ) -= val;
168  lhso.var( s, idc ) -= val;
169  }
170  }
171  }
172  }
173  }
