40  //! Constructor: Quiet output by default, only stuff written to qstr shown.
41  //! Instantiate with str = std::cout for verbose output. Any member function
42  //! can be called by overriding the default stream via the template
43  //! argument, Style, a C-style enum. Note: By default, str == std::clog.
44  //! This is used to initialize str to a local stringstream into which all
45  //! verbose output goes by default, i.e., it will not be shown. This
46  //! solution is chosen instead of trickery with null-streams, as
47  //! boost:formatted output into null-streams caused invalid reads in
48  //! valgrind. This way quiet output (formatted or not) simply goes into a
49  //! local stringstream. In other words, the default argument to str,
50  //! std::clog, is only used to detect whether client code passed a default
51  //! argument or not: if it did not, the string stream is used for verbose
52  //! output, if it did, the specified stream is used for the verbose output.

645  //! \brief Clean up whitespaces and format a long string into multiple lines
646  //! \param[in] str String to format
647  //! \param[in] name String to insert before string to output
648  //! \param[in] indent String to use as identation
649  //! \param[in] width Width in characters to insert newlines for output
650  //! \author J. Bakosi
651  //! \see http://stackoverflow.com/a/6892562
652  //! \see http://stackoverflow.com/a/8362145
653  // TODO A line longer than 'width' will cause a hang!
654  std::string splitLines( std::string str,
655  std::string indent,
656  std::string name = "",
657  std::size_t width = 80 ) const {
658  // remove form feeds, line feeds, carriage returns, horizontal tabs,
659  // vertical tabs, see http://en.cppreference.com/w/cpp/string/byte/isspace
660  str.erase(
661  std::remove_if( str.begin(), str.end(),
662  []( char x ){ return std::isspace( x ) && x != ' '; } ),
663  str.end() );
664  // remove duplicate spaces
665  str.erase(
666  std::unique( str.begin(), str.end(),
667  []( char a, char b ){ return a == b && a == ' '; } ),
668  str.end() );
669  // format str to 'width'-character-long lines with indent
670  str.insert( 0, indent + name );
671  std::size_t currIndex = width - 1;
672  std::size_t sizeToElim;
673  while ( currIndex < str.length() ) {
674  const std::string whitespace = " ";
675  currIndex = str.find_last_of( whitespace, currIndex + 1 );
676  if ( currIndex == std::string::npos ) break;
677  currIndex = str.find_last_not_of( whitespace, currIndex );
678  if ( currIndex == std::string::npos ) break;
679  sizeToElim =
680  str.find_first_not_of( whitespace, currIndex + 1 ) - currIndex - 1;
681  str.replace( currIndex + 1, sizeToElim , "\n" + indent );
682  currIndex += width + indent.length() + 1;
683  }
684  return str;
685  }
