222  //! Extract moment names of requested statistics
223  std::vector< std::string > momentNames( std::function<
224  bool ( const std::vector< tk::ctr::Term >& ) > momentType ) const
225  {
226  std::vector< std::string > names;
227  for (const auto& product : get< tag::stat >()) {
228  if (momentType( product )) {
229  names.emplace_back( std::string() );
230  for (const auto& term : product)
231  names.back() += tk::ctr::Term( term.var, term.field );
232  }
233  }
234  return names;
235  }
