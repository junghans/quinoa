220 void
221 Conductor::aveCost( tk::real c )
222 // *****************************************************************************
223 // Reduction target estimating the average communication cost of merging the
224 // linear system
225 //! \param[in] c Communication cost summed across all PEs. The cost associated
226 //! to a PE is a real number between 0 and 1, defined as the number of mesh
227 //! points the PE does not own, i.e., needs to send to some other PE, divided
228 //! by the total number of points the PE contributes to. The lower the better.
229 //! \details The average, computed here, gives an idea of the average
230 //! communication cost across all PEs, while the standard deviation, computed
231 //! by stdCost(), gives an idea on the expected load imbalance.
232 //! \author J. Bakosi
233 // *****************************************************************************
234 {
235  m_print.diagend( "done" ); // "Partitioning and distributing mesh ...";
236  // Compute average and broadcast it back to all partitioners (PEs)
237  m_avcost = c / CkNumPes();
238  m_partitioner.stdCost( m_avcost );
239 }

283 Conductor::verifybc( CkReductionMsg* msg )
284 // *****************************************************************************
285 // Reduction target initiating verification of the boundary conditions set
286 //! \param[in] msg Serialized and concatenated vectors of BC nodelists
287 //! \details As a final step aggregating all node lists at which LinSysMerger
288 //! will set boundary conditions, this function initiates verification that
289 //! the aggregate boundary conditions to be set match the user's BCs. This
290 //! function is a Charm++ reduction target that is called when all linear
291 //! system merger branches have received from worker chares offering setting
292 //! boundary conditions. This is the first step of the verification. First
293 //! receives the aggregated node list at which LinSysMerger will set boundary
294 //! conditions. Then we do a broadcast to all workers (Performers) to query
295 //! the worker-owned node IDs on which a Dirichlet BC is set by the user (on
296 //! any component of any PDEs integrated by the worker).
297 //! \author J. Bakosi
298 // *****************************************************************************
299 {
300  // Deserialize final BC node list vector set by LinSysMerger
301  PUP::fromMem creator( msg->getData() );
302  creator | m_linsysbc;
303  delete msg;
304 
305  // Issue broadcast querying the BCs set
306  m_performer.requestBCs();

443 void
444 Conductor::evaluateTime()
445 // *****************************************************************************
446 // Evaluate time step and output one-liner report
447 //! \author J. Bakosi
448 // *****************************************************************************
449 {
450  const auto term = g_inputdeck.get< tag::discr, tag::term >();
451  const auto eps = std::numeric_limits< tk::real >::epsilon();
452  const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
453 
454  // if at final stage of time step, finish time step just taken
455  if (m_stage == 1) {
456  // Increase number of iterations taken
457  ++m_it;
458  // Advance physical time to include time step just finished
459  m_t += m_dt;
460  // Truncate the size of last time step
461  if (m_t > term) m_t = term;
462 
463  bool diag = false;
464 
465  // Append diagnostics file at selected times
466  if (!(m_it % g_inputdeck.get< tag::interval, tag::diag >())) {
467  tk::DiagWriter dw( g_inputdeck.get< tag::cmd, tag::io, tag::diag >(),
468  g_inputdeck.get< tag::flformat, tag::diag >(),
469  g_inputdeck.get< tag::prec, tag::diag >(),
470  std::ios_base::app );
471  if (dw.diag( m_it, m_t, m_diag )) diag = true;
472  }
473 
474  if (!(m_it % g_inputdeck.get< tag::interval, tag::tty >())) {
475 
476  // estimate time elapsed and time for accomplishment
477  tk::Timer::Watch ete, eta;
478  const auto& timer = tk::cref_find( m_timer, TimerTag::TIMESTEP );
479  timer.eta( g_inputdeck.get< tag::discr, tag::term >() -
480  g_inputdeck.get< tag::discr, tag::t0 >(),
481  m_t - g_inputdeck.get< tag::discr, tag::t0 >(),
482  g_inputdeck.get< tag::discr, tag::nstep >(),
483  m_it,
484  ete,
485  eta );
486 
487  // Output one-liner
488  m_print << std::setfill(' ') << std::setw(8) << m_it << " "
489  << std::scientific << std::setprecision(6)
490  << std::setw(12) << m_t << " "
491  << m_dt << " "
492  << std::setfill('0')
493  << std::setw(3) << ete.hrs.count() << ":"
494  << std::setw(2) << ete.min.count() << ":"
495  << std::setw(2) << ete.sec.count() << " "
496  << std::setw(3) << eta.hrs.count() << ":"
497  << std::setw(2) << eta.min.count() << ":"
498  << std::setw(2) << eta.sec.count() << " ";
499 
500  // Augment one-liner with output indicators
501  if (!(m_it % g_inputdeck.get<tag::interval,tag::field>())) m_print << 'F';
502  if (diag) m_print << 'D';
503 
504  m_print << '\n';
505  }
506  }
507 
508  // if not final stage of time step or if neither max iterations nor max time
509  // reached, will continue (by telling all linear system merger group
510  // elements to prepare for a new rhs), otherwise finish
511  if (m_stage < 1 || (std::fabs(m_t-term) > eps && m_it < nstep))
512  m_linsysmerger.enable_wait4rhs();
513  else
514  finish();
515 
516  wait4parcom();
517  wait4npar();
518  wait4eval();
519 }
