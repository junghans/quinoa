256 //! \param[in] c Communication cost summed across all PEs. The cost associated

498 void
499 Transporter::evaluateTime()
500 // *****************************************************************************
501 // Evaluate time step and output one-liner report
502 //! \author J. Bakosi
503 // *****************************************************************************
504 {
505  const auto term = g_inputdeck.get< tag::discr, tag::term >();
506  const auto eps = std::numeric_limits< tk::real >::epsilon();
507  const auto nstep = g_inputdeck.get< tag::discr, tag::nstep >();
508 
509  if (m_stage < 1) { // if at half-stage, simply go to next one
510 
511  ++m_stage;
512 
513  } else { // if final stage of time step, finish time step just taken
514 
515  m_stage = 0;
516  // Increase number of iterations taken
517  ++m_it;
518  // Advance physical time to include time step just finished
519  m_t += m_dt;
520 
521  bool diag = false;
522 
523  // Append diagnostics file at selected times
524  if (!(m_it % g_inputdeck.get< tag::interval, tag::diag >())) {
525  tk::DiagWriter dw( g_inputdeck.get< tag::cmd, tag::io, tag::diag >(),
526  g_inputdeck.get< tag::flformat, tag::diag >(),
527  g_inputdeck.get< tag::prec, tag::diag >(),
528  std::ios_base::app );
529  if (dw.diag( m_it, m_t, m_diag )) diag = true;
530  }
531 
532  if (!(m_it % g_inputdeck.get< tag::interval, tag::tty >())) {
533 
534  // estimate time elapsed and time for accomplishment
535  tk::Timer::Watch ete, eta;
536  const auto& timer = tk::cref_find( m_timer, TimerTag::TIMESTEP );
537  timer.eta( g_inputdeck.get< tag::discr, tag::term >() -
538  g_inputdeck.get< tag::discr, tag::t0 >(),
539  m_t - g_inputdeck.get< tag::discr, tag::t0 >(),
540  g_inputdeck.get< tag::discr, tag::nstep >(),
541  m_it,
542  ete,
543  eta );
544 
545  // Output one-liner
546  m_print << std::setfill(' ') << std::setw(8) << m_it << " "
547  << std::scientific << std::setprecision(6)
548  << std::setw(12) << m_t << " "
549  << m_dt << " "
550  << std::setfill('0')
551  << std::setw(3) << ete.hrs.count() << ":"
552  << std::setw(2) << ete.min.count() << ":"
553  << std::setw(2) << ete.sec.count() << " "
554  << std::setw(3) << eta.hrs.count() << ":"
555  << std::setw(2) << eta.min.count() << ":"
556  << std::setw(2) << eta.sec.count() << " ";
557 
558  // Augment one-liner with output indicators
559  if (!(m_it % g_inputdeck.get<tag::interval,tag::field>())) m_print << 'F';
560  if (diag) m_print << 'D';
561 
562  m_print << '\n';
563  }
564  }
565 
566  wait4eval();
567 
568  // if not final stage of time step or if neither max iterations nor max time
569  // reached, will continue (by telling all linear system merger group
570  // elements to prepare for a new rhs), otherwise finish
571  if (m_stage < 1 || (std::fabs(m_t-term) > eps && m_it < nstep))
572  m_linsysmerger.enable_wait4rhs();
573  else
574  finish();
575 }
