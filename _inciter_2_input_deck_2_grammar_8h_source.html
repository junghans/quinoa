74  //! \brief Do general error checking on the differential equation block
75  //! \details This is error checking that generic equation types, such as
76  //! advection-diffusion or the Poisson equation must satisfy. For more
77  //! specific equations, such as Euler or compressible Navier-Stokes, a more
78  //! specialized equation checker does and can do better error checking. See,
79  //! e.g, check_compns.
80  //! \author J. Bakosi
81  template< class eq >
82  struct check_eq : pegtl::action_base< check_eq< eq > > {
83  static void apply( const std::string& value, Stack& stack ) {
84  // Error out if no dependent variable has been selected
85  const auto& depvar = stack.get< tag::param, eq, tag::depvar >();
86  if (depvar.empty() || depvar.size() != neq.get< eq >())
87  tk::grm::Message< Stack, tk::grm::ERROR, tk::grm::MsgKey::NODEPVAR >
88  ( stack, value );
89  // Error out if no number of components has been selected
90  const auto& ncomp = stack.get< tag::component, eq >();
91  if (ncomp.empty() || ncomp.size() != neq.get< eq >())
92  tk::grm::Message< Stack, tk::grm::ERROR, tk::grm::MsgKey::NONCOMP >
93  ( stack, value );
94  // Error out if no test problem has been selected
95  const auto& problem = stack.get< tag::param, eq, tag::problem >();
96  if (problem.empty() || problem.size() != neq.get< eq >())
97  tk::grm::Message< Stack, tk::grm::ERROR, tk::grm::MsgKey::NOINIT >
98  ( stack, value );
99  }
100  };

102  //! \brief Set defaults and do error checking on the compressible
103  //! Navier-Stokes equation block
104  //! \details This is error checking that only the compressible Navier-Stokes
105  //! equation block must satisfy. Besides error checking we also set defaults
106  //! here as this block is called when parsing of a compns...end block has
107  //! just finished.
108  //! \author J. Bakosi
109  template< class eq >
110  struct check_compns : pegtl::action_base< check_compns< eq > > {
111  static void apply( const std::string&, Stack& stack ) {
112  // Set default number of components to 5 (mass, 3 x mom, energy)
113  stack.get< tag::component, eq >().push_back( 5 );
114  // If problem type is not given, default to 'user_defined'
115  auto& problem = stack.get< tag::param, eq, tag::problem >();
116  if (problem.empty()) problem.push_back( ctr::ProblemType::USER_DEFINED );
117  }
118  };
