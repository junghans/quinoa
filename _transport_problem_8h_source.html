58  //! Set initial conditions for dispersion in simple shear flow
59  //! \param[in] coord Mesh node coordinates
60  //! \param[in,out] unk Array of unknowns
61  //! \param[in] e Equation system index, i.e., which advection-diffusion
62  //! system we operate on among the systems of PDEs
63  //! \param[in] offset System offset specifying the position of the system of
64  //! PDEs among other systems
65  //! \param[in] t Physical time
66  template< class eq >
67  static void init( const std::array< std::vector< tk::real >, 3 >& coord,
68  tk::MeshNodes& unk,
69  tk::ctr::ncomp_type e,
70  tk::ctr::ncomp_type ncomp,
71  tk::ctr::ncomp_type offset,
72  tk::real t )
73  {
74  const auto& u0 = g_inputdeck.get< tag::param, eq, tag::u0 >()[e];
75  const auto& lambda = g_inputdeck.get< tag::param, eq, tag::lambda >()[e];
76  const auto& diff =
77  g_inputdeck.get< tag::param, eq, tag::diffusivity >()[e];
78  const tk::real X0 = 7200.0; // x position of source
79  const tk::real t0 = g_inputdeck.get< tag::discr, tag::t0 >(); // initial t
80  const auto& x = coord[0];
81  const auto& y = coord[1];
82  for (ncomp_t c=0; c<ncomp; ++c) {
83  const auto b = 1.0 + lambda[c]*lambda[c]*t*t/12.0;
84  const auto M =
85  4.0*M_PI*t0*std::sqrt( 1.0 + lambda[c]*lambda[c]*t0*t0/12.0 );
86  for (ncomp_t i=0; i<x.size(); ++i) {
87  tk::real a = x[i] - X0 - u0[c]*t - 0.5*lambda[c]*y[i]*t;
88  unk( i, c, offset ) =
89  M * exp( -a*a/(4.0*M_PI*diff[c]*t*b) - y[i]*y[i]/(4.0*diff[c]*t) )
90  / ( 4.0*M_PI*t*std::sqrt(b) );
91  }
92  }
93  }

95  //! Assign prescribed shear velocity to nodes of tetrahedron element
96  //! \param[in] N Element node indices
97  //! \param[in] coord Mesh node coordinates
98  //! \param[in] e Equation system index, i.e., which advection-diffusion
99  //! system we operate on among the systems of PDEs
100  //! \return Velocity assigned to all vertices of a tetrehedron, size:
101  //! ncomp * ndim * nnode = [ncomp][3][4]
102  template< class eq >
103  static std::vector< std::array< std::array< tk::real, 4 >, 3 > >
104  prescribedVelocity( const std::array< std::size_t, 4 >& N,
105  const std::array< std::vector< tk::real >, 3 >& coord,
106  tk::ctr::ncomp_type e,
107  tk::ctr::ncomp_type ncomp )
108  {
109  const auto& u0 = g_inputdeck.get< tag::param, eq, tag::u0 >()[e];
110  const auto& lambda = g_inputdeck.get< tag::param, eq, tag::lambda >()[e];
111  const auto& y = coord[1];
112  std::vector< std::array< std::array< tk::real, 4 >, 3 > > vel( ncomp );
113  for (ncomp_t c=0; c<ncomp; ++c) {
114  std::array< std::array< tk::real, 4 >, 3 > v;
115  v[0][0] = u0[c] + lambda[c]*y[N[0]]; v[1][0] = 0.0; v[2][0] = 0.0;
116  v[0][1] = u0[c] + lambda[c]*y[N[1]]; v[1][1] = 0.0; v[2][1] = 0.0;
117  v[0][2] = u0[c] + lambda[c]*y[N[2]]; v[1][2] = 0.0; v[2][2] = 0.0;
118  v[0][3] = u0[c] + lambda[c]*y[N[3]]; v[1][3] = 0.0; v[2][3] = 0.0;
119  vel[c] = std::move(v);
120  }
121  return vel;
122  }
