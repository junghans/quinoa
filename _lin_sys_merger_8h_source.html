198  //! Receive lower and upper global node IDs all PEs will operate on
199  //! \param[in] pe PE whose bounds being received
200  //! \param[in] lower Lower index of the global rows on my PE
201  //! \param[in] upper Upper index of the global rows on my PE
202  void bounds( int pe, std::size_t lower, std::size_t upper ) {
203  // Store our bounds
204  if (pe == CkMyPe()) {
205  m_lower = lower;
206  m_upper = upper;
207  }
208  // Store inverse of PE-division map stored on all PE
209  m_div[ {lower,upper} ] = pe;
210  // If we have all PEs' bounds, signal the runtime system to continue
211  if (m_div.size() == static_cast<std::size_t>(CkNumPes())) {
212  // Create my PE's lhs matrix distributed across all PEs
213  m_A.create( m_lower*m_ncomp, m_upper*m_ncomp );
214  // Create my PE's rhs and unknown vectors distributed across all PEs
215  m_b.create( m_lower*m_ncomp, m_upper*m_ncomp );
216  m_x.create( m_lower*m_ncomp, m_upper*m_ncomp );
217  // Create linear solver
218  m_solver.create();
219  // Signal back to host that setup of workers can start
220  signal2host_setup( m_host );

243  //! Chares contribute their global row ids
244  //! \param[in] fromch Charm chare array index contribution coming from
245  //! \param[in] row Global mesh point (row) indices contributed
246  //! \note This function does not have to be declared as a Charm++ entry
247  //! method since it is always called by chares on the same PE.
248  void charerow( int fromch, const std::vector< std::size_t >& row ) {
249  // Collect global ids of workers on my PE
250  m_myworker.push_back( fromch );
251  // Store rows owned and pack those to be exported, also build import map
252  // used to test for completion
253  std::map< int, std::set< std::size_t > > exp;
254  for (auto gid : row) {
255  if (gid >= m_lower && gid < m_upper) { // if own
256  m_rowimport[ fromch ].push_back( gid );
257  m_row.insert( gid );
258  } else exp[ pe(gid) ].insert( gid );
259  }
260  // Export non-owned parts to fellow branches that own them
261  m_nperow += exp.size();
262  for (const auto& p : exp) {
263  auto tope = static_cast< int >( p.first );
264  Group::thisProxy[ tope ].
265  addrow( fromch, CkMyPe(), p.second );
266  }
267  if (rowcomplete()) signal2host_row_complete( m_host );
268  }
269  //! Receive global row ids from fellow group branches
270  //! \param[in] fromch Charm chare array index contribution coming from
271  //! \param[in] frompe PE contribution coming from
272  //! \param[in] row Global mesh point (row) indices received
273  void addrow( int fromch, int frompe, const std::set< std::size_t >& row ) {
274  for (auto r : row) {
275  m_rowimport[ fromch ].push_back( r );
276  m_row.insert( r );
277  }
278  Group::thisProxy[ frompe ].recrow();
279  }
280  //! Acknowledge received row ids
281  void recrow() {
282  --m_nperow;

287  //! \param[in] fromch Charm chare array index contribution coming from
288  //! \param[in] gid Global row indices of the vector contributed
289  //! \param[in] sol Portion of the unknown/solution vector contributed
290  //! \note This function does not have to be declared as a Charm++ entry
291  //! method since it is always called by chares on the same PE.
292  void charesol( int fromch,
293  const std::vector< std::size_t >& gid,
294  const MeshNodes& sol )
295  {
296  Assert( gid.size() == sol.nunk(),
297  "Size of solution and row ID vectors must equal" );
298  // Store solution vector nonzero values owned and pack those to be
299  // exported, also build import map used to test for completion
300  std::map< int, std::map< std::size_t, std::vector< tk::real > > > exp;
301  for (std::size_t i=0; i<gid.size(); ++i)
302  if (gid[i] >= m_lower && gid[i] < m_upper) { // if own
303  m_solimport[ fromch ].push_back( gid[i] );
304  m_sol[ gid[i] ] = sol[i];
305  } else {
306  exp[ pe(gid[i]) ][ gid[i] ] = sol[i];
307  }
308  // Export non-owned vector values to fellow branches that own them
309  for (const auto& p : exp) {
310  auto tope = static_cast< int >( p.first );
311  Group::thisProxy[ tope ].addsol( fromch, p.second );
312  }
313  if (solcomplete()) trigger_sol_complete();
314  }

329  //! Chares contribute their matrix nonzero values
330  //! \param[in] fromch Charm chare array index contribution coming from
331  //! \param[in] gid Global row indices of the matrix chunk contributed
332  //! \param[in] psup Points surrounding points using local indices. See also
333  //! tk::genPsup().
334  //! \param[in] lhsd Portion of the left-hand side matrix contributed,
335  //! containing non-zero values (for all scalar components of the equations
336  //! solved) as a sparse matrix diagonal
337  //! \param[in] lhso Portion of the left-hand side matrix contributed,
338  //! containing non-zero values (for all scalar components of the equations
339  //! solved) as a sparse matrix off-diagonal entries in compressed row
340  //! storage format
341  //! \note This function does not have to be declared as a Charm++ entry
342  //! method since it is always called by chares on the same PE.
343  void charelhs( int fromch,
344  const std::vector< std::size_t >& gid,
345  const std::pair< std::vector< std::size_t >,
346  std::vector< std::size_t > >& psup,
347  const tk::MeshNodes& lhsd,
348  const tk::MeshNodes& lhso )
349  {
350  Assert( psup.second.size()-1 == gid.size(),
351  "Number of mesh points and number of global IDs unequal" );
352  Assert( psup.second.size()-1 == lhsd.nunk(),
353  "Number of mesh points and number of diagonals unequal" );
354  Assert( psup.first.size() == lhso.nunk(),
355  "Number of off-diagonals and their number of indices unequal" );
356  // Store matrix nonzero values owned and pack those to be exported, also
357  // build import map used to test for completion
358  std::map< int, std::map< std::size_t,
359  std::map< std::size_t,
360  std::vector< tk::real > > > > exp;
361  for (std::size_t i=0; i<gid.size(); ++i)
362  if (gid[i] >= m_lower && gid[i] < m_upper) { // if own
363  m_lhsimport[ fromch ].push_back( gid[i] );
364  auto& row = m_lhs[ gid[i] ];
365  row[ gid[i] ] += lhsd[i];
366  for (auto j=psup.second[i]+1; j<=psup.second[i+1]; ++j)
367  row[ gid[ psup.first[j] ] ] += lhso[j];
368  } else {
369  auto& row = exp[ pe(gid[i]) ][ gid[i] ];
370  row[ gid[i] ] = lhsd[i];
371  for (auto j=psup.second[i]+1; j<=psup.second[i+1]; ++j)
372  row[ gid[ psup.first[j] ] ] = lhso[j];
373  }
374  // Export non-owned matrix rows values to fellow branches that own them
375  for (const auto& p : exp)
376  Group::thisProxy[ p.first ].addlhs( fromch, p.second );
377  if (lhscomplete()) trigger_lhs_complete();
378  }

436  //! Assert that all global row indices have been received on my PE
437  //! \details The assert consists of three necessary conditions, which
438  //! together comprise the sufficient condition that all global row indices
439  //! have been received owned by this PE.
440  void rowsreceived() {
441  Assert( // 1. have heard from every chare on my PE
442  m_myworker.size() == m_nchare &&
443  // 2. number of rows equals that of the expected on my PE
444  m_row.size() == m_upper-m_lower &&
445  // 3. all fellow PEs have received my row ids contribution
446  m_nperow == 0,
447  // if any of the above is not satisfied, the row ids are

450  // now that the global row ids are complete, build Hypre data from it
451  hyprerow();

521  //! Return processing element for global mesh row id
522  //! \param[in] gid Global mesh point (matrix or vector row) id
523  //! \details First we attempt to the point index in the cache. If that
524  //! fails, we resort to a linear search across the division map. Once the
525  //! PE is found, we store it in the cache, so next time the search is
526  //! quicker. This procedure must find the PE for the id.
527  //! \return PE that owns global row id

529  int p = -1;
530  auto it = m_pe.find( gid );
531  if (it != end(m_pe))
532  p = it->second;
533  else

543  //! \details This does not mean the global row ids on our PE is complete
544  //! (which is tested by an assert in rowsreceived), only that we have done
545  //! our part of receiving contributions from chare array groups storing
546  //! the parts that we own and have sent the parts we do not own to fellow
547  //! PEs, i.e., we have nothing else to export. Only when all other fellow
548  //! branches have received all contributions are the row ids complete on

676  //! Update solution vector in our PE's performers

679  m_x.get( static_cast< int >( (m_upper - m_lower)*m_ncomp ),
680  m_hypreRows.data(),
681  m_hypreSol.data() );

683  // workers that own them

685  std::vector< std::size_t > gid;
686  std::vector< tk::real > sol;
687  for (auto r : w.second) {
688  const auto it = m_sol.find( r );
689  if (it != end(m_sol)) {
690  gid.push_back( it->first );

692  using diff_type = typename decltype(m_hypreSol)::difference_type;
693  auto b = static_cast< diff_type >( i*m_ncomp );
694  auto e = static_cast< diff_type >( (i+1)*m_ncomp );
695  sol.insert( end(sol),
696  std::next( begin(m_hypreSol), b ),
697  std::next( begin(m_hypreSol), e ) );
698  } else
699  Throw( "Can't find global row id " + std::to_string(r) +
700  " to export in solution vector" );
701  }
702  m_worker[ w.first ].updateSolution( gid, sol );
703  }

716  /** @name Host signal calls
717  * \brief These functions signal back to the host via a global reduction
718  * originating from each PE branch
719  * \details Singal calls contribute to a reduction on all branches (PEs)
720  * of LinSysMerger to the host, e.g., inciter::CProxy_Conductor, given by
721  * the template argument HostProxy. The signal functions are overloads on
722  * the specialization, e.g., inciter::CProxy_Conductor, of the
723  * LinSysMerger template. They create Charm++ reduction targets via
724  * creating a callback that invokes the typed reduction client, where
725  * host is the proxy on which the reduction target method, given by the
726  * string followed by "redn_wrapper_", e.g., rowcomplete(), is called
727  * upon completion of the reduction.
728  *
729  * Note that we do not use Charm++'s CkReductionTarget macro here,
730  * but instead explicitly generate the code that that macro would
731  * generate. To explain why, here is Charm++'s CkReductionTarget macro's
732  * definition, given in ckreduction.h:
733  * \code{.cpp}
734  * #define CkReductionTarget(me, method) \
735  * CkIndex_##me::redn_wrapper_##method(NULL)
736  * \endcode
737  * This macro takes arguments 'me' (a class name) and 'method' a member
738  * function of class 'me' and generates the call
739  * 'CkIndex_<class>::redn_wrapper_<method>(NULL)'. With the overloads the
740  * signal2* functions generate, we do the above macro's job for
741  * LinSysMerger specialized by HostProxy, hard-coded here, as well its
742  * reduction target. This is required since
743  * * Charm++'s CkReductionTarget macro's preprocessing happens earlier
744  * than type resolution and the string of the template argument would
745  * be substituted instead of the type specialized (which is not what
746  * we want here), and
747  * * the template argument class, e.g, CProxy_Conductor, is in a
748  * namespace different than that of LinSysMerger. When a new class is
749  * used to specialize LinSysMerger, the compiler will alert that a new
750  * overload needs to be defined.
751  *
752  * \note This simplifies client-code, e.g., inciter::Conductor, which now
753  * requires no explicit book-keeping with counters, etc. Also a reduction
754  * (instead of a direct call to the host) better utilizes the
755  * communication network as computational nodes can send their aggregated
756  * contribution to other nodes on a network instead of all chares sending
757  * their (smaller) contributions to the same host, (hopefully)
758  * implemented using a tree among the PEs.
759  * \see http://charm.cs.illinois.edu/manuals/html/charm++/manual.html,
760  * Sections "Processor-Aware Chare Collections" and "Chare Arrays".
761  * */
762  ///@{
763  //! \brief Signal back to host that the initialization of the row indices of
764  //! the linear system is complete
765  void signal2host_row_complete( const inciter::CProxy_Conductor& host ) {
766  using inciter::CkIndex_Conductor;
767  Group::contribute(
768  CkCallback( CkIndex_Conductor::redn_wrapper_rowcomplete(NULL), host ) );
769  }
770  //! \brief Signal back to host that enabling the SDAG waits for assembling
771  //! the right-hand side is complete and ready for a new advance in time
772  void signal2host_advance( const inciter::CProxy_Conductor& host ) {
773  using inciter::CkIndex_Conductor;
774  Group::contribute(
775  CkCallback( CkIndex_Conductor::redn_wrapper_advance(NULL), host ) );
776  }
777  //! \brief Signal back to host that receiving the inverse PE-division map is
778  //! complete and we are ready for Prformers to start their setup.
779  void signal2host_setup( const inciter::CProxy_Conductor& host ) {
780  using inciter::CkIndex_Conductor;
781  Group::contribute(
782  CkCallback( CkIndex_Conductor::redn_wrapper_setup(NULL), host ) );
783  }
