119 Performer::init( tk::real dt )
120 // *****************************************************************************
121 // Initialize linear system
122 //! \author J. Bakosi
123 // *****************************************************************************
124 {
125  // Set initial conditions for all PDEs
126  for (const auto& eq : g_pdes) eq.initialize( m_coord, m_u, m_t );
127 
128  // Output initial conditions to file (it = 1, time = 0.0)
129  writeFields( m_t );
130 
131  // Send off initial conditions for assembly
132  m_linsysmerger.ckLocalBranch()->charesol( thisIndex, m_gid, m_u );
133 
134  // Call back to Conductor::initcomplete(), signaling that the initialization
135  // is complete and we are now starting time stepping
136  contribute(
137  CkCallback( CkReductionTarget( Conductor, initcomplete ), m_conductor ) );
138 
139  // Compute left-hand side of PDE
140  lhs();
141  // Start advancing PDE in time at time step stage 0
142  advance( 0, dt, m_it, m_t );

304 void
305 Performer::advance( uint8_t stage, tk::real dt, uint64_t it, tk::real t )
306 // *****************************************************************************
307 // Advance equations to next stage in multi-stage time stepping
308 //! \param[in] stage Stage in multi-stage time stepping
309 //! \param[in] dt Size of time step
310 //! \param[in] it Iteration count
311 //! \param[in] t Physical time
312 //! \author J. Bakosi
313 // *****************************************************************************
314 {
315  // Update local copy of time step stage
316  m_stage = stage;
317 
318  // Advance stage in multi-stage time stepping by updating the rhs
319  if (m_stage < 1) {
320 
321  rhs( 0.5, dt, m_u, m_uf );
322 
323  } else {
324 
325  // Update local copy of physical time and iteration count at the final stage
326  m_t = t;
327  m_it = it;
328 
329  rhs( 1.0, dt, m_uf, m_un );
330 
331  }
332 }

334 void
335 Performer::updateSolution( const std::vector< std::size_t >& gid,
336  const std::vector< tk::real >& u )
337 // *****************************************************************************
338 // Update solution vector
339 //! \param[in] gid Global row indices of the vector updated
340 //! \param[in] u Portion of the unknown/solution vector updated
341 //! \author J. Bakosi
342 // *****************************************************************************
343 {
344  auto ncomp = g_inputdeck.get< tag::component >().nprop();
345  Assert( gid.size() * ncomp == u.size(),
346  "Size of row ID vector times the number of scalar components and the "
347  "size of the solution vector must equal" );
348 
349  // Receive update of solution vector
350  for (std::size_t i=0; i<gid.size(); ++i) {
351  auto id = tk::cref_find( m_lid, gid[i] );
352  for (ncomp_t c=0; c<ncomp; ++c) m_un( id, c, 0 ) = u[ i*ncomp+c ];
353  }
354 
355  // Count number of solution nodes updated
356  m_nsol += gid.size();
357 
358  // If all contributions we own have been received, continue by updating a
359  // different solution vector depending on time step stage
360  if (m_nsol == m_gid.size()) {
361 
362  if (m_stage < 1) {
363  m_uf = m_un;
364  } else {
365  m_u = m_un;
366  if (!((m_it+1) % g_inputdeck.get< tag::interval, tag::field >()))
367  writeFields( m_t + g_inputdeck.get< tag::discr, tag::dt >() );
368  }
369 
370  // Prepare for next time step stage
371  m_nsol = 0;
372  // Tell the Charm++ runtime system to call back to Conductor::evaluateTime()
373  // once all Performer chares have received the update. The reduction is done
374  // via creating a callback that invokes the typed reduction client, where
375  // m_conductor is the proxy on which the reduction target method,
376  // evaluateTime(), is called upon completion of the reduction.
377  contribute(
378  CkCallback( CkReductionTarget( Conductor, evaluateTime ), m_conductor ) );
