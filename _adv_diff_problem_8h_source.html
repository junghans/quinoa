60  //! Set initial conditions for dispersion in simple shear flow
61  //! \param[in] deck Input deck
62  //! \param[in] coord Mesh node coordinates
63  //! \param[in,out] unk Array of unknowns
64  //! \param[in] e Equation system index, i.e., which advection-diffusion
65  //! system we operate on among the systems of PDEs
66  //! \param[in] offset System offset specifying the position of the system of
67  //! PDEs among other systems
68  //! \param[in] t Physical time
69  template< class eq >
70  static void init( const ctr::InputDeck& deck,
71  const std::array< std::vector< tk::real >, 3 >& coord,
72  tk::MeshNodes& unk,
73  tk::ctr::ncomp_type e,
74  tk::ctr::ncomp_type ncomp,
75  tk::ctr::ncomp_type offset,
76  tk::real t )
77  {
78  const auto& u0 = deck.get< tag::param, eq, tag::u0 >()[e];
79  const auto& lambda = deck.get< tag::param, eq, tag::lambda >()[e];
80  const auto& diff = deck.get< tag::param, eq, tag::diffusivity >()[e];
81  const tk::real X0 = 7200.0; // x position of source
82  const tk::real t0 = deck.get< tag::discr, tag::t0 >(); // initial t
83  const auto& x = coord[0];
84  const auto& y = coord[1];
85  for (ncomp_t c=0; c<ncomp; ++c) {
86  const auto b = 1.0 + lambda[c]*lambda[c]*t*t/12.0;
87  const auto M =
88  4.0*M_PI*t0*std::sqrt( 1.0 + lambda[c]*lambda[c]*t0*t0/12.0 );
89  for (ncomp_t i=0; i<x.size(); ++i) {
90  tk::real a = x[i] - X0 - u0[c]*t - 0.5*lambda[c]*y[i]*t;
91  unk( i, c, offset ) =
92  M * exp( -a*a/(4.0*M_PI*diff[c]*t*b) - y[i]*y[i]/(4.0*diff[c]*t) )
93  / ( 4.0*M_PI*t*std::sqrt(b) );
94  }
95  }
96  }

98  //! Assign prescribed shear velocity to nodes of tetrahedron element
99  //! \param[in] A Vertex index of tetrahedron element
100  //! \param[in] B Vertex index of tetrahedron element
101  //! \param[in] C Vertex index of tetrahedron element
102  //! \param[in] D Vertex index of tetrahedron element
103  //! \param[in] coord Mesh node coordinates
104  //! \param[in] e Equation system index, i.e., which advection-diffusion
105  //! system we operate on among the systems of PDEs
106  //! \return Velocity assigned to all vertices of a tetrehedron, size:
107  //! ncomp * ndim * nnode = [ncomp][3][4]
108  template< class eq >
109  static std::vector< std::array< std::array< tk::real, 4 >, 3 > >
110  velocity( const ctr::InputDeck& deck,
111  std::size_t A, std::size_t B, std::size_t C, std::size_t D,
112  const std::array< std::vector< tk::real >, 3 >& coord,
113  tk::ctr::ncomp_type e,
114  tk::ctr::ncomp_type ncomp )
115  {
116  const auto& u0 = deck.get< tag::param, eq, tag::u0 >()[e];
117  const auto& lambda = deck.get< tag::param, eq, tag::lambda >()[e];
118  const auto& y = coord[1];
119  std::vector< std::array< std::array< tk::real, 4 >, 3 > > vel( ncomp );
120  for (ncomp_t c=0; c<ncomp; ++c) {
121  std::array< std::array< tk::real, 4 >, 3 > v;
122  v[0][0] = u0[c] + lambda[c]*y[A]; v[1][0] = 0.0; v[2][0] = 0.0;
123  v[0][1] = u0[c] + lambda[c]*y[B]; v[1][1] = 0.0; v[2][1] = 0.0;
124  v[0][2] = u0[c] + lambda[c]*y[C]; v[1][2] = 0.0; v[2][2] = 0.0;
125  v[0][3] = u0[c] + lambda[c]*y[D]; v[1][3] = 0.0; v[2][3] = 0.0;
126  vel[c] = std::move(v);
127  }
128  return vel;
129  }

132  //! \return Problem type as enum
133  static ctr::ProblemType type() noexcept
134  { return ctr::ProblemType::SHEAR_DIFF; }
