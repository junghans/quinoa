152 std::unordered_map< std::size_t, std::array< tk::real, 3 > >
153 ExodusIIMeshReader::readNodes( const std::array< std::size_t, 2 >& ext ) const
154 // *****************************************************************************
155 // Read coordinates of a number of mesh nodes from ExodusII file
156 //! \param[in] ext Extents of element ids whose connectivity to read, both
157 //! inclusive
158 //! \param[inout] coord Unordered map of node coordinates associated to ids
159 //! \author J. Bakosi
160 // *****************************************************************************
161 {
162  auto num = ext[1] - ext[0] + 1;
163  std::vector< tk::real > px( num ), py( num ), pz( num );
164 
165  ErrChk(
166  ex_get_n_coord(
167  m_inFile, static_cast<int64_t>(ext[0])+1, static_cast<int64_t>(num),
168  px.data(), py.data(), pz.data() ) == 0,
169  "Failed to read coordinates of nodes [" + std::to_string(ext[0]) +
170  "..." + std::to_string(ext[1]) + "] from ExodusII file: " +
171  m_filename );
172 
173  std::unordered_map< std::size_t, std::array< tk::real, 3 > > coord;
174  for (std::size_t p=0; p<num; ++p)
175  coord.emplace( ext[0]+p, std::array<tk::real,3>{{px[p],py[p],pz[p]}} );
176  return coord;
177 }

179 std::size_t
180 ExodusIIMeshReader::readElemBlockIDs()
181 // *****************************************************************************
182 // Read element block IDs from ExodusII file
183 //! \return Total number of nodes in mesh
184 //! \author J. Bakosi
185 // *****************************************************************************
186 {
187  // Read ExodusII file header
188  auto nnode = readHeader();
189 
190  std::vector< int > eid( m_neblk );
191 
192  // Read element block ids
193  ErrChk( ex_get_elem_blk_ids( m_inFile, eid.data()) == 0,
194  "Failed to read element block ids from ExodusII file: " +
195  m_filename );
196 
197  // Fill element block ID vector
198  for (auto id : eid) {
199  char eltype[MAX_STR_LENGTH+1];
200  int n, nnpe, nattr;
201 
202  // Read element block information
203  ErrChk(
204  ex_get_elem_block( m_inFile, id, eltype, &n, &nnpe, &nattr ) == 0,
205  "Failed to read element block information from ExodusII file: " +
206  m_filename );
207 
208  // Store ExodusII element block ID
209  m_eid.push_back( id );
210 
211  // Store ExodusII element block ID mapped to tk::ExoElemType enum, and
212  // number of elements per block mapped to tk::ExoElemType enum
213  if (nnpe == 4) { // tetrahedra
214  m_eidt[ static_cast<std::size_t>(ExoElemType::TET) ] = id;
215  m_nel[ static_cast<std::size_t>(ExoElemType::TET) ] = n;
216  } else if (nnpe == 3) { // triangles
217  m_eidt[ static_cast<std::size_t>(ExoElemType::TRI) ] = id;
218  m_nel[ static_cast<std::size_t>(ExoElemType::TRI) ] = n;
219  }
220  }
221 
222  return nnode;
223 }

226 void
227 ExodusIIMeshReader::readAllElements( UnsMesh& mesh )
228 // *****************************************************************************
229 // Read all element blocks and mesh connectivity from ExodusII file
230 //! \param[inout] mesh Unstructured mesh object to store mesh in
231 //! \author J. Bakosi
232 // *****************************************************************************
233 {
234  // Read element block ids
235  readElemBlockIDs();
236 
237  for (auto id : m_eid) {
238  char eltype[MAX_STR_LENGTH+1];
239  int nelem, nnpe, nattr;
240 
241  // Read element block information
242  ErrChk(
243  ex_get_elem_block( m_inFile, id, eltype, &nelem, &nnpe, &nattr ) == 0,
244  "Failed to read element block information from ExodusII file: " +
245  m_filename );
246 
247  // Read element connectivity
248  auto connectsize = static_cast< std::size_t >( nelem*nnpe );
249  if (nnpe == 4) { // tetrahedra
250 
251  mesh.tettag().resize( connectsize, { 1 } );
252  std::vector< int > inpoel( connectsize );
253  ErrChk(
254  ex_get_elem_conn( m_inFile, id, inpoel.data() ) == 0,
255  "Failed to read " + std::string(eltype) + " element connectivity from "
256  "ExodusII file: " + m_filename );
257  for (auto n : inpoel)
258  mesh.tetinpoel().push_back( static_cast< std::size_t >( n ) );
259 
260  } else if (nnpe == 3) { // triangles
261 
262  mesh.tritag().resize( connectsize, { 1 } );
263  std::vector< int > inpoel( connectsize );
264  ErrChk(
265  ex_get_elem_conn( m_inFile, id, inpoel.data() ) == 0,
266  "Failed to read " + std::string(eltype) + " element connectivity from "
267  "ExodusII file: " + m_filename );
268  for (auto n : inpoel)
269  mesh.triinpoel().push_back( static_cast< std::size_t >( n ) );
270 
271  }
272  }
273 
274  // Shift node IDs to start from zero
275  shiftToZero( mesh.triinpoel() );
276  shiftToZero( mesh.tetinpoel() );
277 }

283 // *****************************************************************************
284 // Read element connectivity of a single mesh cell from ExodusII file
285 //! \param[in] id Element id whose connectivity to read
286 //! \param[in] elemtype Element type
287 //! \param[inout] conn Connectivity vector to push to
288 //! \note Must be preceded by a call to readElemBlockIDs()
289 //! \author J. Bakosi
290 // *****************************************************************************
291 {
292  Assert( static_cast< std::size_t >(
293  std::accumulate(begin(m_eidt), end(m_eidt), 0) ) != -m_nnpe.size(),
294  "A call to ExodusIIMeshReader::readElement() must be preceded by a "
295  "call to ExodusIIMeshReader::readElemBlockIDs()" );
296 
297  auto bid = static_cast< std::size_t >( elemtype );
298 
299  std::vector< int > c( m_nnpe[bid] );
300 
301  // Read element connectivity from file
302  ErrChk(
303  ex_get_n_elem_conn(

305  "Failed to read element connectivity of element " + std::to_string( id ) +
306  " from block " + std::to_string(m_eidt[bid]) + " from ExodusII file: " +
307  m_filename );
308 
309  // Put in element connectivity using zero-based node indexing
310  for (auto i : c) conn.push_back( static_cast<std::size_t>(i)-1 );
311 }

352 std::map< int, std::vector< std::size_t > >
353 ExodusIIMeshReader::readSidesets()
354 // *****************************************************************************
355 // Read node list of all side sets from ExodusII file
356 //! \return Elem and side lists mapped to side set ids
357 //! \author J. Bakosi
358 // *****************************************************************************
359 {
360  // Read ExodusII file header (fills m_neset)
361  readHeader();
362 
363  // Node lists mapped to side set ids
364  std::map< int, std::vector< std::size_t > > side;
365 
366  if (m_neset > 0) {
367  // Read all side set ids from file
368  std::vector< int > ids( m_neset );
369  ErrChk( ex_get_side_set_ids( m_inFile, ids.data() ) == 0,
370  "Failed to read side set ids from ExodusII file: " + m_filename );
371  // Read in node list for all side sets
372  for (auto i : ids) {
373  int nface, nnode;
374  // Read number of faces and number of distribution factors in side set i
375  ErrChk( ex_get_side_set_param( m_inFile, i, &nface, &nnode ) == 0,
376  "Failed to read side set " + std::to_string(i) + " parameters "
377  "from ExodusII file: " + m_filename );
378  // Read number of nodes in side set i (overwrite nnode)
379  ErrChk( ex_get_side_set_node_list_len( m_inFile, i, &nnode ) == 0,
380  "Failed to read side set " + std::to_string(i) + " node list "
381  "length from ExodusII file: " + m_filename );
382  Assert(nnode > 0, "Number of nodes = 0 in side set" + std::to_string(i));
383  std::vector< int > df( static_cast< std::size_t >( nface ) );
384  std::vector< int > nodes( static_cast< std::size_t >( nnode ) );
385  // Read in node list for side set i
386  ErrChk( ex_get_side_set_node_list( m_inFile, i, df.data(), nodes.data() )
387  == 0, "Failed to read node list of side set " +
388  std::to_string(i) + " from ExodusII file: " +
389  m_filename );
390  // Make node list unique
391  tk::unique( nodes );
392  // Store 0-based node ID list as std::size_t vector instead of ints
393  auto& list = side[ i ];
394  for (auto&& n : nodes) list.push_back( static_cast<std::size_t>(n-1) );
395  }
396  }
397 
398  return side;
399 }
